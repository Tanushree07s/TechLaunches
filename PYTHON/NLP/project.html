  def rule_based_chatbot(user_input):
      user_input = user_input.lower()

      if "hello" in user_input or "hi" in user_input:
          return "Hello! How can I help you today?"
      elif "bye" in user_input or "goodbye" in user_input:
          return "Goodbye! Have a great day!"
      elif "how are you" in user_input:
          return "I'm doing great! Thanks for asking!"
      elif "your name" in user_input:
          return "I'm a rule-based chatbot!"
      else:
          return "I'm not sure how to respond to that."

  # Example interaction
  while True:
      user_input = input("You: ")
      if user_input.lower() in ["exit", "quit"]:
          print("Bot: Goodbye!")
          break
      response = rule_based_chatbot(user_input)
      print("Bot:", response)

Solution: Retrival-Based Chatbot


  from sklearn.feature_extraction.text import TfidfVectorizer
  from sklearn.metrics.pairwise import cosine_similarity

  # Define known questions and answers
  qa_pairs = {
      "hello": "Hello! How can I assist you?",
      "how are you": "I'm doing great! Thanks for asking!",
      "what is your name": "I'm a retrieval-based chatbot.",
      "bye": "See you later!"
  }

  # Prepare TF-IDF
  vectorizer = TfidfVectorizer()
  questions = list(qa_pairs.keys())
  tfidf_matrix = vectorizer.fit_transform(questions)

  def retrieval_based_chatbot(user_input):
      user_tfidf = vectorizer.transform([user_input])
      similarities = cosine_similarity(user_tfidf, tfidf_matrix)
      best_match_idx = similarities.argmax()

      if similarities[0, best_match_idx] < 0.3:
          return "Sorry, I don't understand that."
      return qa_pairs[questions[best_match_idx]]

  # Example interaction
  while True:
      user_input = input("You: ")
      if user_input.lower() in ["exit", "quit"]:
          print("Bot: Goodbye!")
          break
      response = retrieval_based_chatbot(user_input.lower())
      print("Bot:", response)
